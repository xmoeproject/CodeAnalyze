184 #if !defined(MBEDTLS_ECP_WINDOW_SIZE)
185 /*
186  * Maximum "window" size used for point multiplication.
187  * Default: 6.
188  * Minimum value: 2. Maximum value: 7.
189  *
190  * Result is an array of at most ( 1 << ( MBEDTLS_ECP_WINDOW_SIZE - 1 ) )
191  * points used for point multiplication. This value is directly tied to EC
192  * peak memory usage, so decreasing it by one should roughly cut memory usage
193  * by two (if large curves are in use).
194  *
195  * Reduction in size may reduce speed, but larger curves are impacted first.
196  * Sample performances (in ECDHE handshakes/s, with FIXED_POINT_OPTIM = 1):
197  * w-size: 6 5 4 3 2
198  * 521 145 141 135 120 97
199  * 384 214 209 198 177 146
200  * 256 320 320 303 262 226
201 
202  * 224 475 475 453 398 342
203  * 192 640 640 633 587 476
204  */
205 #define MBEDTLS_ECP_WINDOW_SIZE 6
206 #endif /* MBEDTLS_ECP_WINDOW_SIZE */
207 
208 #if !defined(MBEDTLS_ECP_FIXED_POINT_OPTIM)
209 /*
210  * Trade memory for speed on fixed-point multiplication.
211  *
212  * This speeds up repeated multiplication of the generator (that is, the
213  * multiplication in ECDSA signatures, and half of the multiplications in
214  * ECDSA verification and ECDHE) by a factor roughly 3 to 4.
215  *
216  * The cost is increasing EC peak memory usage by a factor roughly 2.
217  *
218  * Change this value to 0 to reduce peak memory usage.
219  */
220 #define MBEDTLS_ECP_FIXED_POINT_OPTIM 1
221 #endif /* MBEDTLS_ECP_FIXED_POINT_OPTIM */
