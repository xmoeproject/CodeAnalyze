42 BINARY_TO_BCD(USHORT Value)
43 {
44  USHORT Result;
45 
46  Result = (Value / 1000) << 12;
47  Value %= 1000;
48  Result |= (Value / 100) << 8;
49  Value %= 100;
50  Result |= (Value / 10) << 4;
51  Value %= 10;
52  Result |= Value;
53 
54  return Result;

59 BCD_TO_BINARY(USHORT Value)
60 {
61  USHORT Result;
62 
63  Result = Value & 0xF;
64  Value >>= 4;
65  Result += (Value & 0xF) * 10;
66  Value >>= 4;
67  Result += (Value & 0xF) * 100;
68  Value >>= 4;
