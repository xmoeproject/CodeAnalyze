45 {
46  int num = 0;
47 
48 #if BITS_PER_LONG == 64
49  if ((word & 0xffffffff) == 0) {
50  num += 32;
51  word >>= 32;
52  }
53 #endif
54  if ((word & 0xffff) == 0) {
55  num += 16;
56  word >>= 16;
57  }
58  if ((word & 0xff) == 0) {
59  num += 8;

61  }

63  num += 4;
64  word >>= 4;
65  }
66  if ((word & 0x3) == 0) {
67  num += 2;
68  word >>= 2;
69  }
70  if ((word & 0x1) == 0)

182 static inline int ffs(int x)
183 {
184  int r = 1;
185 
186  if (!x)
187  return 0;
188  if (!(x & 0xffff)) {
189  x >>= 16;
190  r += 16;
191  }
192  if (!(x & 0xff)) {
193  x >>= 8;
194  r += 8;
195  }
196  if (!(x & 0xf)) {
197  x >>= 4;
198  r += 4;
199  }
200  if (!(x & 3)) {
201  x >>= 2;
202  r += 2;
203  }
204  if (!(x & 1)) {
205  x >>= 1;
206  r += 1;
207  }

313 static inline unsigned long hweight32(unsigned long w)
314 {
315  unsigned int res = (w & 0x55555555) + ((w >> 1) & 0x55555555);
316  res = (res & 0x33333333) + ((res >> 2) & 0x33333333);
317  res = (res & 0x0F0F0F0F) + ((res >> 4) & 0x0F0F0F0F);
318  res = (res & 0x00FF00FF) + ((res >> 8) & 0x00FF00FF);
319  return (res & 0x0000FFFF) + ((res >> 16) & 0x0000FFFF);
320 }
