160 static inline BYTE d3drm_color_component(float c)
161 {
162  if (c <= 0.0f)
163  return 0u;
164  if (c >= 1.0f)
165  return 0xffu;
166  return floor(c * 255.0f);
167 }

170 {
171  *color = RGBA_MAKE(d3drm_color_component(r), d3drm_color_component(g),
172  d3drm_color_component(b), d3drm_color_component(a));
