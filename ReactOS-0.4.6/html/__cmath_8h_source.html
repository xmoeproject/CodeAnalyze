194 #define _STLP_MATH_INLINE2PX(__type,func,cfunc) \
195  inline __type func (__type x, __type *y) { \

197  tmp1 = _STLP_CMATH_FUNC_NAMESPACE::cfunc(__STATIC_CAST(double, x), &tmp2); \
