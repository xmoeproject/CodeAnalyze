97 /*
98  * Global Lock acquire/release code
99  *
100  * Note: Handles case where the FACS pointer is null
101  */
102 #define ACPI_ACQUIRE_GLOBAL_LOCK(FacsPtr, Acq) \
103 do { \
104  asm("orl %2,%2;" \
105  "jz 2f;" \
106  "leal 16(%2),%2;" \
107  "1:" \
108  "movl (%2),%%eax;" \
109  "movl %%eax,%%edx;" \
110  "andl %3,%%edx;" \
111  "btsl $0x1,%%edx;" \
112  "adcl $0x0,%%edx;" \
113  "lock; cmpxchgl %%edx,(%2);" \
114  "jnz 1b;" \
115  "cmpb $0x3,%%dl;" \
116  "sbbl %%eax,%%eax;" \
117  "2:" \
118  :"=a"(Acq):"a"(0xFF),"c"(FacsPtr),"i"(~1L):"edx");\
119 } while(0)
120 
121 #define ACPI_RELEASE_GLOBAL_LOCK(FacsPtr, Acq) \
122 do { \
123  asm("orl %2,%2;" \
124  "jz 2f;" \
125  "leal 16(%2),%2;" \
126  "1:" \
127  "movl (%2),%%eax;" \
128  "movl %%eax,%%edx;" \
129  "andl %3,%%edx;" \
130  "lock; cmpxchgl %%edx,(%2);" \
131  "jnz 1b;" \
132  "cmpb $0x3,%%dl;" /* FIXME: this is pointless */ \
133  "andl $0x1,%%eax;" \
134  "2:" \
135  :"=a"(Acq):"a"(0),"c"(FacsPtr),"i"(~3L):"edx");\
136 } while(0)
