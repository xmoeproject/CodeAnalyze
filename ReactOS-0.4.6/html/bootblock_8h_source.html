144  * MBR (Master Boot Record) --

148 /*
149  * Layout of boot records:
150  *
151  * Byte range Use Description
152  * ---------- --- -----------
153  *
154  * 0 - 2 FMP JMP xxx, NOP
155  * 3 - 10 FP OEM Name
156  *
157  * 11 - 61 FMP FAT12/16 BPB
158  * Whilst not strictly necessary for MBR,
159  * GRUB reserves this area
160  *
161  * 11 - 89 P FAT32 BPB
162  * (are we ever going to boot off this?)
163  *
164  *
165  * 62 - 217 FMP Boot code
166  *
167  * 90 - 217 P FAT32 boot code
168  *
169  * 218 - 223 M Win95b/98/me "drive time"
170  * http://www.geocities.com/thestarman3/asm/mbr/95BMEMBR.htm#MYST
171  * only changed if all 6 bytes are 0
172  *
173  * 224 - 436 FMP boot code (continued)
174  *
175  * 437 - 439 M WinNT/2K/XP MBR "boot language"
176  * http://www.geocities.com/thestarman3/asm/mbr/Win2kmbr.htm
177  * not needed by us
178  *
179  * 400 - 439 MP NetBSD: mbr_bootsel
180  *
181  * 440 - 443 M WinNT/2K/XP Drive Serial Number (NT DSN)
182  * http://www.geocities.com/thestarman3/asm/mbr/Win2kmbr.htm
183  *
184  * 444 - 445 FMP bootcode or unused
185  * NetBSD: mbr_bootsel_magic
186  *
187  * 446 - 509 M partition table
188  *
189  * 510 - 511 FMP magic number (0xAA55)
190  *
191  * Use:
192  * ----
193  * F Floppy boot sector
194  * M Master Boot Record
195  * P Partition Boot record
196  *
197  */
198

439 #define ALPHA_BOOT_BLOCK_CKSUM(bb,cksum) \
440  do { \
441  const struct alpha_boot_block *_bb = (bb); \
442  uint64_t _cksum; \
443  int _i; \
444  \
445  _cksum = 0; \
446  for (_i = 0; \
447  _i < (sizeof _bb->bb_data / sizeof _bb->bb_data[0]); \
448  _i++) \
449  _cksum += _bb->bb_data[_i]; \
450  *(cksum) = _cksum; \
451  } while (/*CONSTCOND*/ 0)
