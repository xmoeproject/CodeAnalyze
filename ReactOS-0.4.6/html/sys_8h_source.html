296 #ifndef SYS_ARCH_INC
297 #define SYS_ARCH_INC(var, val) do { \
298  SYS_ARCH_DECL_PROTECT(old_level); \
299  SYS_ARCH_PROTECT(old_level); \
300  var += val; \
301  SYS_ARCH_UNPROTECT(old_level); \
302  } while(0)
303 #endif /* SYS_ARCH_INC */
