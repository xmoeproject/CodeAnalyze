164 static inline char *d3dcompiler_strdup(const char *string)
165 {
166  char *copy;
167  SIZE_T len;
168 
169  if (!string)
170  return NULL;
171 
172  len = strlen(string);
173  copy = d3dcompiler_alloc(len + 1);
174  if (copy)

176  return copy;
