722 static __inline void get_raid0_offset(UINT64 off, UINT64 stripe_length, UINT16 num_stripes, UINT64* stripeoff, UINT16* stripe) {
723  UINT64 initoff, startoff;
724 
725  startoff = off % (num_stripes * stripe_length);
726  initoff = (off / (num_stripes * stripe_length)) * stripe_length;
