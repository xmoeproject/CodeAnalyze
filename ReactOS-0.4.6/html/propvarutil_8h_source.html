91 {
92  ppropvar->vt = VT_BOOL;
93  ppropvar->boolVal = fVal ? VARIANT_TRUE : VARIANT_FALSE;

97 inline HRESULT InitPropVariantFromString(PCWSTR psz, PROPVARIANT *ppropvar)
98 {
99  HRESULT hres;
100 
101  hres = SHStrDupW(psz, &ppropvar->pwszVal);
102  if(SUCCEEDED(hres))
103  ppropvar->vt = VT_LPWSTR;
104  else
105  PropVariantInit(ppropvar);
106 
107  return hres;
108 }
