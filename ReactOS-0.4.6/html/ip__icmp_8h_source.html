110 /*
111  * Lower bounds on packet lengths for various types.
112  * For the error advice packets must first insure that the
113  * packet is large enough to contain the returned ip header.
114  * Only then can we do the check to see if 64 bits of packet
115  * data have been returned, since we need to check the returned
116  * ip header length.
117  */
118 #define ICMP_MINLEN 8 /* abs minimum */
119 #define ICMP_TSLEN (8 + 3 * sizeof (n_time)) /* timestamp */
120 #define ICMP_MASKLEN 12 /* address mask */
121 #define ICMP_ADVLENMIN (8 + sizeof (struct ip) + 8) /* min */
122 #ifndef _IP_VHL
123 #define ICMP_ADVLEN(p) (8 + ((p)->icmp_ip.ip_hl << 2) + 8)
124  /* N.B.: must separately check that ip_hl >= 5 */
125 #else
126 #define ICMP_ADVLEN(p) (8 + (IP_VHL_HL((p)->icmp_ip.ip_vhl) << 2) + 8)
127  /* N.B.: must separately check that header length >= 5 */
128 #endif
