53 /*
54  * The root inode is the root of the file system. Inode 0 can't be used for
55  * normal purposes and historically bad blocks were linked to inode 1, thus
56  * the root inode is 2. (Inode 1 is no longer used for this purpose, however
57  * numerous dump tapes make this assumption, so we are stuck with it).
58  */
59 #define ROOTINO ((ino_t)2)
60 
61 /*
62  * The Whiteout inode# is a dummy non-zero inode number which will
63  * never be allocated to a real file. It is used as a place holder
64  * in the directory entry which has been tagged as a DT_W entry.
65  * See the comments about ROOTINO above.
66  */
67 #define WINO ((ino_t)1)

69 /*
70  * A dinode contains all the meta-data associated with a UFS file.
71  * This structure defines the on-disk format of a dinode. Since
72  * this structure describes an on-disk structure, all its fields
73  * are defined by types with precise widths.
74  */
75 
76 #define NXADDR 2
77 #define NDADDR 12 /* Direct addresses in inode. */
78 #define NIADDR 3 /* Indirect addresses in inode. */
79 
80 struct ufs1_dinode {
81  u_int16_t di_mode; /* 0: IFMT, permissions; see below. */
82  int16_t di_nlink; /* 2: File link count. */
83  union {
84  u_int16_t oldids[2]; /* 4: Ffs: old user and group ids. */
85  u_int32_t inumber; /* 4: Lfs: inode number. */
86  } di_u;
87  u_int64_t di_size; /* 8: File byte count. */
88  int32_t di_atime; /* 16: Last access time. */
89  int32_t di_atimensec; /* 20: Last access time. */
90  int32_t di_mtime; /* 24: Last modified time. */
91  int32_t di_mtimensec; /* 28: Last modified time. */
92  int32_t di_ctime; /* 32: Last inode change time. */
93  int32_t di_ctimensec; /* 36: Last inode change time. */
94  int32_t di_db[NDADDR]; /* 40: Direct disk blocks. */
95  int32_t di_ib[NIADDR]; /* 88: Indirect disk blocks. */
96  u_int32_t di_flags; /* 100: Status flags (chflags). */
97  u_int32_t di_blocks; /* 104: Blocks actually held. */
98  int32_t di_gen; /* 108: Generation number. */
99  u_int32_t di_uid; /* 112: File owner. */
100  u_int32_t di_gid; /* 116: File group. */
101  int32_t di_spare[2]; /* 120: Reserved; currently unused */
102 };

104 struct ufs2_dinode {
105  u_int16_t di_mode; /* 0: IFMT, permissions; see below. */
106  int16_t di_nlink; /* 2: File link count. */
107  u_int32_t di_uid; /* 4: File owner. */
108  u_int32_t di_gid; /* 8: File group. */
109  u_int32_t di_blksize; /* 12: Inode blocksize. */
110  u_int64_t di_size; /* 16: File byte count. */
111  u_int64_t di_blocks; /* 24: Bytes actually held. */
112  int64_t di_atime; /* 32: Last access time. */
113  int64_t di_mtime; /* 40: Last modified time. */
114  int64_t di_ctime; /* 48: Last inode change time. */
115  int64_t di_birthtime; /* 56: Inode creation time. */
116  int32_t di_mtimensec; /* 64: Last modified time. */
117  int32_t di_atimensec; /* 68: Last access time. */
118  int32_t di_ctimensec; /* 72: Last inode change time. */
119  int32_t di_birthnsec; /* 76: Inode creation time. */
120  int32_t di_gen; /* 80: Generation number. */
121  u_int32_t di_kernflags; /* 84: Kernel flags. */
122  u_int32_t di_flags; /* 88: Status flags (chflags). */
123  int32_t di_extsize; /* 92: External attributes block. */
124  int64_t di_extb[NXADDR];/* 96: External attributes block. */
125  int64_t di_db[NDADDR]; /* 112: Direct disk blocks. */
126  int64_t di_ib[NIADDR]; /* 208: Indirect disk blocks. */
127  int64_t di_spare[3]; /* 232: Reserved; currently unused */
128 };

130 /*
131  * The di_db fields may be overlaid with other information for
132  * file types that do not have associated disk storage. Block
133  * and character devices overlay the first data block with their
134  * dev_t value. Short symbolic links place their path in the
135  * di_db area.
136  */
137 #define di_inumber di_u.inumber
138 #define di_ogid di_u.oldids[1]
139 #define di_ouid di_u.oldids[0]
140 #define di_rdev di_db[0]
141 #define MAXSYMLINKLEN_UFS1 ((NDADDR + NIADDR) * sizeof(int32_t))
142 #define MAXSYMLINKLEN_UFS2 ((NDADDR + NIADDR) * sizeof(int64_t))
143

145  ((ip)->i_ump->um_fstype == UFS1) ? \
146  MAXSYMLINKLEN_UFS1 : MAXSYMLINKLEN_UFS2
