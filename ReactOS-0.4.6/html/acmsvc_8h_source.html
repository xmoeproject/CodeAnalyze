164 #define ACPI_ACQUIRE_GLOBAL_LOCK(FacsPtr, Acq) __asm \
165 { \
166  __asm mov eax, 0xFF \
167  __asm mov ecx, FacsPtr \
168  __asm or ecx, ecx \
169  __asm jz exit_acq \
170  __asm lea ecx, [ecx].GlobalLock \
171  \
172  __asm acq10: \
173  __asm mov eax, [ecx] \
174  __asm mov edx, eax \
175  __asm and edx, 0xFFFFFFFE \
176  __asm bts edx, 1 \
177  __asm adc edx, 0 \
178  __asm lock cmpxchg dword ptr [ecx], edx \
179  __asm jnz acq10 \
180  \
181  __asm cmp dl, 3 \
182  __asm sbb eax, eax \
183  \
184  __asm exit_acq: \
185  __asm mov Acq, al \
186 }

188 #define ACPI_RELEASE_GLOBAL_LOCK(FacsPtr, Pnd) __asm \
189 { \
190  __asm xor eax, eax \
191  __asm mov ecx, FacsPtr \
192  __asm or ecx, ecx \
193  __asm jz exit_rel \
194  __asm lea ecx, [ecx].GlobalLock \
195  \
196  __asm Rel10: \
197  __asm mov eax, [ecx] \
