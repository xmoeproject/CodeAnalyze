91 /*
92  * Global Lock acquire/release code
93  *
94  * Note: Handles case where the FACS pointer is null
95  */
96 #define ACPI_ACQUIRE_GLOBAL_LOCK(FacsPtr, Acq) __asm \
97 { \
98  __asm mov eax, 0xFF \
99  __asm mov ecx, FacsPtr \
100  __asm or ecx, ecx \
101  __asm jz exit_acq \
102  __asm lea ecx, [ecx].GlobalLock \
103  \
104  __asm acq10: \
105  __asm mov eax, [ecx] \
106  __asm mov edx, eax \
107  __asm and edx, 0xFFFFFFFE \
108  __asm bts edx, 1 \
109  __asm adc edx, 0 \
110  __asm lock cmpxchg dword ptr [ecx], edx \
111  __asm jnz acq10 \
112  \
113  __asm cmp dl, 3 \
114  __asm sbb eax, eax \
115  \
116  __asm exit_acq: \
117  __asm mov Acq, al \
118 }
119 
120 #define ACPI_RELEASE_GLOBAL_LOCK(FacsPtr, Pnd) __asm \
121 { \
122  __asm xor eax, eax \
123  __asm mov ecx, FacsPtr \
124  __asm or ecx, ecx \
125  __asm jz exit_rel \
126  __asm lea ecx, [ecx].GlobalLock \
127  \
128  __asm Rel10: \
129  __asm mov eax, [ecx] \
130  __asm mov edx, eax \
131  __asm and edx, 0xFFFFFFFC \
132  __asm lock cmpxchg dword ptr [ecx], edx \
133  __asm jnz Rel10 \
134  \
135  __asm cmp dl, 3 \
136  __asm and eax, 1 \
137  \
138  __asm exit_rel: \
139  __asm mov Pnd, al \
140 }
141 #endif /* __REACTOS__ */
