84  * The following code partially allows libschily to access FILE * as fast as

86  * This makes it possible to implement js_printf() from libschily aprox.
87  * 33% faster than printf() from libc on Solaris. To do this, we
88  * partially unhide the FILE structure in a 64 bit environment on Solaris
89  * to allow to run putc_unlocked() as a marcro.
90  *
91  * If you believe you can do this on onther platforms, send a note.
92  */
93 #if defined(__SVR4) && defined(__sun) && defined(_LP64)
94 #ifndef _SCHILY_TYPES_H
95 #include <schily/types.h> /* Needed for ssize_t */
96 #endif

98 /*
99  * This is how the 64 bit FILE * begins on Solaris.
100  */
101 struct SCHILY__FILE_TAG {
102  unsigned char *_ptr; /* next character from/to here in buffer */
103  unsigned char *_base; /* the buffer */
104  unsigned char *_end; /* the end of the buffer */
105  ssize_t _cnt; /* number of available characters in buffer */
106 };

109  ? __filbuf((FILE *)p) \

112 #define getc_unlocked(p) __getc_unlocked((struct SCHILY__FILE_TAG *)p)

114 #define __putc_unlocked(x, p) (--(p)->_cnt < 0 \
115  ? __flsbuf((x), (FILE *)(p)) \
116  : (int)(*(p)->_ptr++ = \
117  (unsigned char) (x)))

119 #define putc_unlocked(x, p) __putc_unlocked(x, (struct SCHILY__FILE_TAG *)p)
