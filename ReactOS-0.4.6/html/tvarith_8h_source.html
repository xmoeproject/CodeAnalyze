31 static inline void timevaladd(struct timeval *tv_io, long ms) {
32  struct timeval tmp;
33  assert(ms>=0);
34  tmp= *tv_io;
35  tmp.tv_usec += (ms%1000)*1000;
36  tmp.tv_sec += ms/1000;
37  if (tmp.tv_usec >= 1000000) { tmp.tv_sec++; tmp.tv_usec -= 1000000; }
38  *tv_io= tmp;
39 }
