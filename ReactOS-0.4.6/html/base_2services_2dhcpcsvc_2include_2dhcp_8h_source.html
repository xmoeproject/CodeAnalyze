57 struct dhcp_packet {
58  u_int8_t op; /* Message opcode/type */
59  u_int8_t htype; /* Hardware addr type (see net/if_types.h) */
60  u_int8_t hlen; /* Hardware addr length */
61  u_int8_t hops; /* Number of relay agent hops from client */
62  u_int32_t xid; /* Transaction ID */
63  u_int16_t secs; /* Seconds since client started looking */
64  u_int16_t flags; /* Flag bits */
65  struct in_addr ciaddr; /* Client IP address (if already in use) */
66  struct in_addr yiaddr; /* Client IP address */
67  struct in_addr siaddr; /* IP address of next server to talk to */
68  struct in_addr giaddr; /* DHCP relay agent IP address */
69  unsigned char chaddr[16]; /* Client hardware address */
70  char sname[DHCP_SNAME_LEN]; /* Server name */
71  char file[DHCP_FILE_LEN]; /* Boot filename */
72  unsigned char options[DHCP_OPTION_LEN];
73  /* Optional parameters
74  (actual length dependent on MTU). */
75 };
