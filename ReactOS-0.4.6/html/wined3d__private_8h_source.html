248 /* float_16_to_32() and float_32_to_16() (see implementation in
249  * surface_base.c) convert 16 bit floats in the FLOAT16 data type
250  * to standard C floats and vice versa. They do not depend on the encoding
251  * of the C float, so they are platform independent, but slow. On x86 and
252  * other IEEE 754 compliant platforms the conversion can be accelerated by
253  * bit shifting the exponent and mantissa. There are also some SSE-based
254  * assembly routines out there.
255  *
256  * See GL_NV_half_float for a reference of the FLOAT16 / GL_HALF format
257  */
258 static inline float float_16_to_32(const unsigned short *in)
259 {
260  const unsigned short s = ((*in) & 0x8000u);
261  const unsigned short e = ((*in) & 0x7c00u) >> 10;
262  const unsigned short m = (*in) & 0x3ffu;
263  const float sgn = (s ? -1.0f : 1.0f);
264 
265  if(e == 0) {
266  if(m == 0) return sgn * 0.0f; /* +0.0 or -0.0 */
267  else return sgn * powf(2, -14.0f) * ((float)m / 1024.0f);
268  } else if(e < 31) {
269  return sgn * powf(2, (float)e - 15.0f) * (1.0f + ((float)m / 1024.0f));
270  } else {
271  if(m == 0) return sgn * INFINITY;
272  else return NAN;
273  }
274 }

276 static inline float float_24_to_32(DWORD in)
277 {
278  const float sgn = in & 0x800000u ? -1.0f : 1.0f;
279  const unsigned short e = (in & 0x780000u) >> 19;
280  const unsigned int m = in & 0x7ffffu;
281 
282  if (e == 0)
283  {
284  if (m == 0) return sgn * 0.0f; /* +0.0 or -0.0 */
285  else return sgn * powf(2, -6.0f) * ((float)m / 524288.0f);
286  }
287  else if (e < 15)
288  {
289  return sgn * powf(2, (float)e - 7.0f) * (1.0f + ((float)m / 524288.0f));
290  }
291  else
292  {
293  if (m == 0) return sgn * INFINITY;
294  else return NAN;
295  }
296 }
